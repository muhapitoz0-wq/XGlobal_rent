        function displayGear(items) {
            const grid = document.getElementById('gearGrid');
            grid.innerHTML = items.map(item => `
                <div class="card" data-type="${item.type}">
                    <div class="card-img">
                        <span class="badge">Available</span>
                        [ ${item.name} IMAGE ]
                    </div>
                    <div class="card-content">
                        <div class="card-title">${item.name}</div>
                        <div class="card-acc">${item.acc}</div>
                        <div class="card-footer">
                            <div class="price">$${item.rate}<span>/day</span></div>
                            <button class="btn-add" onclick="updateQuote()">Add to Quote</button>
                        </div>
                    </div>
                </div>
            `).join('');
        }

        function filterGear() {
            const val = document.getElementById('searchInput').value.toLowerCase();
            const filtered = inventory.filter(item => 
                item.name.toLowerCase().includes(val) || 
                item.type.toLowerCase().includes(val)
            );
            displayGear(filtered);
        }

        function filterCategory(cat, el) {
            // Update UI tabs
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            el.classList.add('active');

            if(cat === 'All') {
                displayGear(inventory);
            } else {
                const filtered = inventory.filter(item => item.type === cat);
                displayGear(filtered);
            }
        }

        function updateQuote() {
            count++;
            document.getElementById('quoteCount').innerText = `Quote (${count})`;
        }

        // Initialize
        displayGear(inventory);
